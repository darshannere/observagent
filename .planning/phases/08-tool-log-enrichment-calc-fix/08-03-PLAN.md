CHANGE 2 — Update createRow() to use .log-row-main wrapper and add summary element:

Replace the body of createRow() with:

  function createRow(event) {
    const row = document.createElement('div');
    row.className = 'log-row';
    if (event.hook_type === 'PreToolUse') row.classList.add('in-progress');

    // Wrap main content in a horizontal flex row
    const mainLine = document.createElement('div');
    mainLine.className = 'log-row-main';

    const toolEl = document.createElement('span');
    toolEl.className = 'tool-name';
    const isError = event.exit_status !== null && event.exit_status !== undefined && event.exit_status !== 0;
    toolEl.textContent = isError ? event.tool_name + '\u2014 error' : event.tool_name;
    if (isError) row.classList.add('error');

    const tsEl = document.createElement('span');
    tsEl.className = 'ts';
    tsEl.textContent = formatTs(event.timestamp);

    const durEl = document.createElement('span');
    durEl.className = 'duration ' + latencyClass(event.duration_ms);
    durEl.textContent = event.duration_ms !== null ? formatDuration(event.duration_ms) : '\u2026';

    mainLine.appendChild(toolEl);
    mainLine.appendChild(tsEl);
    mainLine.appendChild(durEl);
    row.appendChild(mainLine);

    // Second line: tool_summary (only if present — PreToolUse branch only)
    if (event.tool_summary) {
      const summaryEl = document.createElement('div');
      summaryEl.className = 'tool-summary';
      summaryEl.textContent = event.tool_summary;
      summaryEl.title = event.tool_summary;
      row.appendChild(summaryEl);
    }

    return { row, durEl };
  }

IMPORTANT: The .error class and isError logic already set the .error class on row — keep this unchanged. The durEl is still returned because appendRow() PostToolUse branch uses it.
