**JavaScript (inline `<script>` before `</body>`):**

All JS is self-contained — no external libraries. Write these functions:

```javascript
// --- State ---
const agentSections = new Map();   // session_id -> container element (div inside details)
const inProgressTimers = new Map(); // tool_call_id -> { intervalId, row, timerEl }

// --- Helpers ---
function formatTs(tsMs) {
  return new Date(tsMs).toLocaleTimeString('en-US', { hour12: false });
}

function formatDuration(ms) {
  if (ms === null || ms === undefined) return '—';
  return ms < 1000 ? `${ms}ms` : `${(ms / 1000).toFixed(1)}s`;
}

function latencyClass(ms) {
  if (ms === null || ms === undefined) return '';
  if (ms < 500) return 'green';
  if (ms <= 2000) return 'yellow';
  return 'red';
}

// --- Auto-scroll ---
const logContainer = document.getElementById('tool-log');
let userScrolledUp = false;
logContainer.addEventListener('scroll', () => {
  const atBottom = logContainer.scrollHeight - logContainer.scrollTop <= logContainer.clientHeight + 60;
  userScrolledUp = !atBottom;
});

function scrollToBottom() {
  if (!userScrolledUp) {
    requestAnimationFrame(() => { logContainer.scrollTop = logContainer.scrollHeight; });
  }
}

// --- Agent sections ---
function getAgentSection(sessionId) {
  if (agentSections.has(sessionId)) return agentSections.get(sessionId);

  const details = document.createElement('details');
  details.open = true;
  details.className = 'agent-section';
  details.dataset.sessionId = sessionId;

  const summary = document.createElement('summary');
  summary.textContent = `Agent ${sessionId.slice(0, 8)}`;

  const container = document.createElement('div');
  details.appendChild(summary);
  details.appendChild(container);
  logContainer.appendChild(details);
  agentSections.set(sessionId, container);
  scrollToBottom();
  return container;
}

// --- Row creation ---
function createRow(event) {
  const row = document.createElement('div');
  row.className = 'log-row';
  if (event.hook_type === 'PreToolUse') row.classList.add('in-progress');

  const toolEl = document.createElement('span');
  toolEl.className = 'tool-name';
  // Error rows: show "toolname — error" if exit_status is non-null non-zero
  const isError = event.exit_status !== null && event.exit_status !== undefined && event.exit_status !== 0;
  toolEl.textContent = isError ? `${event.tool_name} — error` : event.tool_name;
  if (isError) row.classList.add('error');

  const tsEl = document.createElement('span');
  tsEl.className = 'ts';
  tsEl.textContent = formatTs(event.timestamp);

  const durEl = document.createElement('span');
  durEl.className = `duration ${latencyClass(event.duration_ms)}`;
  durEl.textContent = event.duration_ms !== null ? formatDuration(event.duration_ms) : '…';

  row.appendChild(toolEl);
  row.appendChild(tsEl);
  row.appendChild(durEl);
  return { row, durEl };
}

// --- Live event rendering ---
function appendRow(event) {
  if (event.hook_type === 'PreToolUse') {
    const { row, durEl } = createRow(event);
    if (event.tool_call_id) {
      const startMs = event.timestamp;
      const intervalId = setInterval(() => {
        const elapsed = Date.now() - startMs;
        durEl.textContent = formatDuration(elapsed) + '…';
        durEl.className = 'duration'; // no color while in progress
      }, 100);
      // Cleanup orphaned timers after 60s (stuck tool protection)
      const killTimer = setTimeout(() => {
        if (inProgressTimers.has(event.tool_call_id)) {
          clearInterval(intervalId);
          inProgressTimers.delete(event.tool_call_id);
          durEl.textContent = '60s+ (stuck)';
          durEl.className = 'duration red';
          row.classList.remove('in-progress');
        }
      }, 60_000);
      inProgressTimers.set(event.tool_call_id, { intervalId, killTimer, row, durEl });
    }
    getAgentSection(event.session_id).appendChild(row);
    scrollToBottom();

  } else if (event.hook_type === 'PostToolUse') {
    if (event.tool_call_id && inProgressTimers.has(event.tool_call_id)) {
      // Update existing in-progress row in-place
      const pending = inProgressTimers.get(event.tool_call_id);
      clearInterval(pending.intervalId);
      clearTimeout(pending.killTimer);
      inProgressTimers.delete(event.tool_call_id);

      pending.durEl.textContent = formatDuration(event.duration_ms);
      pending.durEl.className = `duration ${latencyClass(event.duration_ms)}`;
      pending.row.classList.remove('in-progress');

      const isError = event.exit_status !== null && event.exit_status !== undefined && event.exit_status !== 0;
      if (isError) {
        pending.row.classList.add('error');
        pending.row.querySelector('.tool-name').textContent = `${event.tool_name} — error`;
        showToast(`${event.tool_name} failed`);
      }
    } else {
      // No matching PreToolUse row (pre-existing event or server restart) — render standalone
      const { row } = createRow(event);
      getAgentSection(event.session_id).appendChild(row);
      scrollToBottom();
    }
  }
}

// --- Historical hydration (no timers — duration already known) ---
function renderHistoricalRow(event) {
  // For history, skip standalone PreToolUse rows that have a matching PostToolUse
  // Simple approach: render all rows as final (no timers)
  const { row } = createRow(event);
  getAgentSection(event.session_id).appendChild(row);
}

async function hydrate() {
  try {
    const res = await fetch('/api/events');
    if (!res.ok) return;
    const events = await res.json();
    // stmtAll returns DESC; reverse to chronological for display
    const ordered = events.slice().reverse();
    for (const e of ordered) {
      renderHistoricalRow(e);
    }
    scrollToBottom();
  } catch (e) {
    console.warn('[hydrate] failed:', e.message);
  }
}

// --- SSE subscription ---
function subscribeSSE() {
  const es = new EventSource('/events');
  es.onmessage = (e) => {
    const event = JSON.parse(e.data);
    if (event.type === 'connected') return; // skip initial handshake
    appendRow(event);
  };
  es.onerror = () => {
    console.warn('[sse] connection lost — browser will auto-reconnect');
  };
}

// --- Toast notifications ---
let toastContainer = null;
function ensureToastContainer() {
  if (!toastContainer) {
    toastContainer = document.createElement('div');
    toastContainer.style.cssText = 'position:fixed;bottom:16px;right:16px;display:flex;flex-direction:column;gap:8px;z-index:9999;';
    document.body.appendChild(toastContainer);
  }
  return toastContainer;
}
function showToast(message) {
  const container = ensureToastContainer();
  const toast = document.createElement('div');
  toast.style.cssText = `
    background:#1f1f1f;border:1px solid var(--red);
    color:var(--text);padding:10px 14px;border-radius:6px;
    font-size:12px;font-family:var(--mono);
    opacity:0;transition:opacity 0.2s;max-width:280px;
    word-break:break-word;
  `;
  toast.textContent = message;
  container.appendChild(toast);
  requestAnimationFrame(() => { toast.style.opacity = '1'; });
  setTimeout(() => {
    toast.style.opacity = '0';
    setTimeout(() => toast.remove(), 200);
  }, 5000);
}

// --- Init ---
document.addEventListener('DOMContentLoaded', async () => {
  await hydrate();
  subscribeSSE();
});
```
