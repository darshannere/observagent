Correct approach — use only display:none in the HTML, JS sets it to 'flex':
```html
<div id="replay-banner" style="display:none; ...">
```
Then in JS: `document.getElementById('replay-banner').style.display = 'flex';`

**Step 3 — Add export helper functions** in the script section:
```javascript
function toCsvRow(values) {
  return values.map(v => {
    const s = (v === null || v === undefined) ? '' : String(v);
    return (s.includes(',') || s.includes('"') || s.includes('\n'))
      ? '"' + s.replace(/"/g, '""') + '"' : s;
  }).join(',');
}

function triggerDownload(content, filename, mimeType) {
  const blob = new Blob([content], { type: mimeType });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}

async function exportSession(sessionId, format) {
  const data = await fetch('/api/sessions/' + sessionId + '/export').then(r => r.json());
  const { session, events } = data;
  const project = (session && session.project_name) || 'unknown';
  const date = ((session && session.last_event_ts) || '').slice(0, 10) || new Date().toISOString().slice(0, 10);
  const safeName = project.replace(/[^a-zA-Z0-9_-]/g, '_');
  const filename = 'observagent_' + safeName + '_' + date + '.' + format;

  if (format === 'jsonl') {
    const content = events.map(e => JSON.stringify(Object.assign({ session_id: sessionId, project_name: project }, e))).join('\n');
    triggerDownload(content, filename, 'application/jsonl');
  } else {
    const headers = ['session_id', 'project_name', 'tool_name', 'timestamp', 'duration_ms', 'exit_status'];
    const rows = events.map(e => [sessionId, project, e.tool_name, e.timestamp, e.duration_ms, e.exit_status]);
    const content = [headers, ...rows].map(toCsvRow).join('\n');
    triggerDownload(content, filename, 'text/csv');
  }
}
```
