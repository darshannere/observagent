Add a helper function addColumnIfNotExists and call it inside initDb after the existing db.exec block.

The helper:
```javascript
function addColumnIfNotExists(db, table, col, typeDef) {
  const exists = db.prepare(`PRAGMA table_info(${table})`).all().find(c => c.name === col);
  if (!exists) db.exec(`ALTER TABLE ${table} ADD COLUMN ${col} ${typeDef}`);
}
```

Inside initDb, after the existing db.exec table creation block, add:
```javascript
addColumnIfNotExists(db, 'session_cost', 'project_name', "TEXT NOT NULL DEFAULT ''");
db.exec(`CREATE INDEX IF NOT EXISTS idx_session_cost_project ON session_cost(project_name, last_event_ts DESC)`);
console.log('[db] project_name column and index ready');
```

Do NOT use "ALTER TABLE session_cost ADD COLUMN IF NOT EXISTS ..." — that syntax does not exist in SQLite 3.51.2 and throws a syntax error. The PRAGMA check pattern is idempotent and runs safely on every server start.

function extractProjectName(rawRecords) {
  for (const r of rawRecords) {
    if (r.cwd && typeof r.cwd === 'string' && r.cwd.length > 1) {
      return basename(r.cwd); // e.g. '/Users/x/claude/observagent' -> 'observagent'
    }
  }
  return 'unknown';
}

**C. Add startup backfill** inside startJsonlWatcher, after the existing JSONL discovery loops complete (after the `console.log('[cost] subagent JSONL discovery complete')` line):
```javascript
// Backfill project_name for sessions loaded before Phase 5
try {
  const emptyRows = db.prepare("SELECT session_id FROM session_cost WHERE project_name = '' AND agent_id = ''").all();
  if (emptyRows.length > 0) {
    console.log(`[cost] backfilling project_name for ${emptyRows.length} sessions...`);
    const updateStmt = db.prepare("UPDATE session_cost SET project_name = ? WHERE session_id = ?");
    const dirEntries = await readdir(PROJECTS_DIR, { withFileTypes: true }).catch(() => []);
    const projectDirs = dirEntries.filter(e => e.isDirectory()).map(e => join(PROJECTS_DIR, e.name));
    for (const row of emptyRows) {
      let found = false;
      for (const dir of projectDirs) {
        const candidate = join(dir, row.session_id + '.jsonl');
        try {
          await stat(candidate);
          const recs = await parseJsonlFile(candidate);
          const pn = extractProjectName(recs);
          updateStmt.run(pn, row.session_id);
          found = true;
          break;
        } catch { continue; }
      }
      if (!found) updateStmt.run('unknown', row.session_id);
    }
    console.log('[cost] project_name backfill complete');
  }
} catch {
  // Backfill errors must never crash the server
}
```

stat and readdir are already imported from 'node:fs/promises'. No new imports needed.
