    **Add JavaScript** for the cost panel. Add a new `<script>` block (or append to existing one) with:

    ```javascript
    // ── Cost Panel ──────────────────────────────────────────────────
    // State
    let costState = {
      sessions: [],
      todayTotal: 0,
      thresholds: { budget_threshold_usd: null, ctx_fill_threshold_pct: null }
    };

    // Format cost: always 6 decimal places, $ prefix, no label
    function formatCost(usd) {
      return '$' + Number(usd || 0).toFixed(6);
    }

    // Format token count with K/M suffix for readability
    function formatTokens(n) {
      n = n || 0;
      if (n >= 1_000_000) return (n / 1_000_000).toFixed(1) + 'M';
      if (n >= 1_000)     return (n / 1_000).toFixed(0) + 'K';
      return String(n);
    }

    // Find the most recently active session (highest updated_at / most tokens)
    function getActiveSession(sessions) {
      if (!sessions.length) return null;
      return sessions[0]; // sessions sorted by updated_at DESC from /api/cost
    }

    // Render cost panel from costState
    function renderCostPanel() {
      const active = getActiveSession(costState.sessions);

      // Session cost
      document.getElementById('session-cost').textContent = active
        ? formatCost(active.total_cost_usd)
        : '$0.000000';

      // Today's total
      document.getElementById('today-cost').textContent = formatCost(costState.todayTotal);

      // Token breakdown from active session
      const tokIn    = active?.input_tokens      || 0;
      const tokOut   = active?.output_tokens     || 0;
      const tokCR    = active?.cache_read_tokens || 0;
      const tokCW    = active ? (active.cache_write_5m || 0) + (active.cache_write_1h || 0) : 0;
      document.getElementById('tok-input').textContent       = formatTokens(tokIn);
      document.getElementById('tok-output').textContent      = formatTokens(tokOut);
      document.getElementById('tok-cache-read').textContent  = formatTokens(tokCR);
      document.getElementById('tok-cache-write').textContent = formatTokens(tokCW);

      // Context fill — input + cache_read + cache_write represents context usage
      // We receive contextFillPct from SSE events; for hydration, compute from most recent session
      // Use the stored pct if available, else leave at previous value
      const pct = costState.contextFillPct ?? 0;
      document.getElementById('ctx-pct').textContent = pct + '%';
      const bar = document.getElementById('ctx-bar');
      bar.style.width = pct + '%';
      bar.className = 'progress-fill' +
        (pct >= 95 ? ' danger' : pct >= 80 ? ' warning' : '');

      // Model breakdown: group all sessions by model, sum cost
      const modelMap = {};
      for (const s of costState.sessions) {
        if (!s.model) continue;
        if (!modelMap[s.model]) modelMap[s.model] = { cost: 0, sessions: 0 };
        modelMap[s.model].cost += s.total_cost_usd || 0;
        modelMap[s.model].sessions++;
      }
      const breakdownEl = document.getElementById('model-breakdown');
      if (Object.keys(modelMap).length) {
        breakdownEl.innerHTML = Object.entries(modelMap)
          .sort((a, b) => b[1].cost - a[1].cost)
          .map(([model, d]) =>
            `<div><span class="model-name">${model.replace('claude-', '')}</span> — ${formatCost(d.cost)} (${d.sessions} session${d.sessions > 1 ? 's' : ''})</div>`
          ).join('');
      } else {
        breakdownEl.innerHTML = '<div style="color:var(--muted)">No sessions yet</div>';
      }

      // Alert banner — check thresholds
      renderAlertBanner(active);
    }

    function renderAlertBanner(activeSession) {
      const alertEl = document.getElementById('cost-alert');
      const { budget_threshold_usd, ctx_fill_threshold_pct } = costState.thresholds;
      const alerts = [];

      if (budget_threshold_usd !== null && activeSession) {
        if ((activeSession.total_cost_usd || 0) >= budget_threshold_usd) {
          alerts.push(`Session cost ${formatCost(activeSession.total_cost_usd)} exceeded budget $${budget_threshold_usd.toFixed(2)}`);
        }
      }
      if (ctx_fill_threshold_pct !== null) {
        const pct = costState.contextFillPct ?? 0;
        if (pct >= ctx_fill_threshold_pct) {
          alerts.push(`Context ${pct}% — exceeded ${ctx_fill_threshold_pct}% threshold`);
        }
      }

      if (alerts.length) {
        alertEl.textContent = alerts.join(' · ');
        alertEl.style.display = 'block';
      } else {
        alertEl.style.display = 'none';
      }
    }

    // Hydrate cost panel on page load
    async function hydrateCost() {
      try {
        const [costData, configData] = await Promise.all([
          fetch('/api/cost').then(r => r.json()),
          fetch('/api/config').then(r => r.json()),
        ]);
        costState.sessions   = costData.sessions  || [];
        costState.todayTotal = costData.todayTotal || 0;
        costState.thresholds = configData;
        // Set contextFillPct from most recent session if available
        // (SSE will update this live; for initial load, approximate from token counts)
        renderCostPanel();

        // Populate threshold inputs with persisted values
        const bi = document.getElementById('budget-input');
        const ci = document.getElementById('ctx-threshold-input');
        if (configData.budget_threshold_usd !== null)   bi.value = configData.budget_threshold_usd;
        if (configData.ctx_fill_threshold_pct !== null) ci.value = configData.ctx_fill_threshold_pct;
      } catch (e) {
        console.warn('[cost] hydration failed:', e.message);
      }
    }

    // Handle live cost_update SSE events
    function handleCostUpdate(msg) {
      // msg shape: { type, sessionId, cost, tokens, contextFillPct, model, ts }
      // Update or insert the session in costState.sessions
      const idx = costState.sessions.findIndex(s => s.session_id === msg.sessionId);
      const updated = {
        session_id:        msg.sessionId,
        model:             msg.model,
        input_tokens:      msg.tokens.input,
        output_tokens:     msg.tokens.output,
        cache_read_tokens: msg.tokens.cacheRead,
        cache_write_5m:    msg.tokens.cacheWrite,
        cache_write_1h:    0,
        total_cost_usd:    msg.cost,
        last_event_ts:     new Date(msg.ts).toISOString(),
      };
      if (idx >= 0) {
        costState.sessions[idx] = updated;
      } else {
        costState.sessions.unshift(updated); // prepend — most recent first
      }
      costState.contextFillPct = msg.contextFillPct;
      renderCostPanel();
    }

    // Wire threshold inputs: POST to /api/config on change (debounced 600ms)
    function wireThresholdInputs() {
      let debounceTimer;
      function saveThresholds() {
        clearTimeout(debounceTimer);
        debounceTimer = setTimeout(async () => {
          const bi = document.getElementById('budget-input');
          const ci = document.getElementById('ctx-threshold-input');
          const body = {};
          body.budget_threshold_usd   = bi.value  ? parseFloat(bi.value)  : null;
          body.ctx_fill_threshold_pct = ci.value  ? parseInt(ci.value, 10) : null;
          costState.thresholds = {
            budget_threshold_usd:   body.budget_threshold_usd,
            ctx_fill_threshold_pct: body.ctx_fill_threshold_pct,
          };
          try {
            await fetch('/api/config', {
              method:  'POST',
              headers: { 'Content-Type': 'application/json' },
              body:    JSON.stringify(body),
            });
          } catch (e) {
            console.warn('[cost] config save failed:', e.message);
          }
          renderAlertBanner(getActiveSession(costState.sessions));
        }, 600);
      }
      document.getElementById('budget-input').addEventListener('input', saveThresholds);
      document.getElementById('ctx-threshold-input').addEventListener('input', saveThresholds);
    }

    // Hook into existing SSE message handler
    // Find the existing evtSource.onmessage or evtSource.addEventListener handler in the file
    // Add a branch: if (msg.type === 'cost_update') { handleCostUpdate(msg); return; }
    // This MUST be integrated into the EXISTING SSE handler — do not create a second EventSource.

    // Initialization
    document.addEventListener('DOMContentLoaded', () => {
      hydrateCost();
      wireThresholdInputs();
    });
    ```
