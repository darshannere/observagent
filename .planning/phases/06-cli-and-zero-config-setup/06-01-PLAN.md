**Change 2 — Port and listen log (lines ~30-35):**
Replace:
```javascript
fastify.listen({ port: 4999, host: '127.0.0.1' }, (err) => {
  if (err) {
    console.error('[server] startup error:', err.message);
    process.exit(1);
  }
  console.log('[server] ObservAgent listening on port 4999');
  startJsonlWatcher(db).catch(e => console.error('[cost] watcher startup error:', e.message));
});
```
With:
```javascript
const PORT = parseInt(process.env.PORT ?? '4999', 10);
fastify.listen({ port: PORT, host: '127.0.0.1' }, (err) => {
  if (err) {
    console.error('[server] startup error:', err.message);
    process.exit(1);
  }
  console.log(`[server] ObservAgent listening on port ${PORT}`);
  startJsonlWatcher(db).catch(e => console.error('[cost] watcher startup error:', e.message));
});
```

The fallback `?? '4999'` preserves the existing default so that running `node server.js` directly (without env var) continues to work on port 4999.
